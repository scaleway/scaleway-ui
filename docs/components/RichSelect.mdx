import { useState, useCallback } from 'react'

# RichSelect

## Uncontrolled

RichSelect can be used as an [uncontrolled component](https://reactjs.org/docs/uncontrolled-components.html).

<Playground>
  <RichSelect>
    <RichSelect.Option value="a">Option A</RichSelect.Option>
    <RichSelect.Option value="b">Option B</RichSelect.Option>
  </RichSelect>
</Playground>

## Controlled

Most of the time, you need a [controlled component](https://reactjs.org/docs/forms.html).

<Playground>
  {() => {
    const [value, setValue] = useState({ value: 'a', label: 'Option A' })
    const handleChange = useCallback(value => setValue(value), [])
    return (
      <RichSelect value={value} onChange={handleChange}>
        <RichSelect.Option value="a">Option A</RichSelect.Option>
        <RichSelect.Option value="b">Option B</RichSelect.Option>
      </RichSelect>
    )
  }}
</Playground>

## Multi

Most of the time, you need a [controlled component](https://reactjs.org/docs/forms.html).

<Playground>
  {() => {
    const [value, setValue] = useState()
    const handleChange = useCallback(value => setValue(value), [])
    return (
      <RichSelect value={value} onChange={handleChange} isMulti required>
        <RichSelect.Option value="a">Option A</RichSelect.Option>
        <RichSelect.Option value="b">Option B</RichSelect.Option>
        <RichSelect.Option value="c">Option C</RichSelect.Option>
        <RichSelect.Option value="d">Option D</RichSelect.Option>
      </RichSelect>
    )
  }}
</Playground>
