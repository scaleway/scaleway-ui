# SwitchButton

<Description>
  SwitchButton is an advanced version of Switch component made out of two
  different buttons.
</Description>

<Canvas>
  <Story name="Basic">
    <ControlValue value="a">
      {({ value, onChange }) => (
        <Grid>
          <Row>
            <Col xsmall={3}>
              <SwitchButton
                name="switch-0"
                value="a"
                checked={value === 'a'}
                onChange={e => onChange(e.currentTarget.value)}
              >
                This is A
              </SwitchButton>
            </Col>
            <Col xsmall={3}>
              <SwitchButton
                name="switch-1"
                value="b"
                checked={value === 'b'}
                onChange={e => onChange(e.currentTarget.value)}
              >
                This is B
              </SwitchButton>
            </Col>
          </Row>
        </Grid>
      )}
    </ControlValue>
  </Story>
</Canvas>
