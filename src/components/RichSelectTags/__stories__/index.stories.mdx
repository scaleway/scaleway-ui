# RichSelectTags

## Controlled

RichSelectTags only work as [controlled component](https://reactjs.org/docs/forms.html).

<Canvas>
  <Story name="Basic">
    <ControlValue value={[]}>
      {({ onChange: onChangeFormValue }) => (
        <ControlValue value={{ label: 'Two', value: 'two' }}>
          {({ value, onChange: onChangeSelectValue }) => {
            const props = {
              ctaText: 'Add',
              initialTags: ['one'],
              name: 'options',
              onChange: onChangeFormValue,
              onChangeSelectValue,
              options: [
                { label: 'One', value: 'one' },
                { label: 'Two', value: 'two' },
                { label: 'Three', value: 'three' },
                { label: 'Four', value: 'four' },
                { label: 'Five', value: 'five' },
              ],
              placeholder: 'Select an option in the list',
              value,
            }
            return <RichSelectTags {...props} />
          }}
        </ControlValue>
      )}
    </ControlValue>
  </Story>
</Canvas>
