import {
  ArgsTable,
  Canvas,
  Description,
  Meta,
  Story,
} from '@storybook/addon-docs'
import Icon, { icons } from '..'

<Meta title="Components/Icon" component={Icon} />

# Icon

<Description>Allow you to display an svg icon</Description>

<Canvas>
  <Story name="Basic">
    <Icon name="auto-fix" size={32} color="primary" />
  </Story>
</Canvas>

## Name

<Description>Set desired icon using `name` property.</Description>

<Canvas>
  <Story name="Name">
    {icons.map(name => (
      <Box my={1} key={name} style={{ fontSize: '1.3em' }}>
        <Icon name={name} /> {name}
      </Box>
    ))}
  </Story>
</Canvas>

## Size

<Description>Set size using `size` property.</Description>

<Canvas>
  <Story name="Size">
    <Boxer mr={2} display="inline-block">
      <Icon name="eye" size={40} />
      <Icon name="eye" size={50} />
      <Icon name="eye" size={60} />
    </Boxer>
  </Story>
</Canvas>

## Color

<Description>
  Set color using `color` property. (doesn't work with `compute, iot, network,
  storage, tools` icons which are already colored)
</Description>

<Canvas>
  <Story name="Color">
    <Boxer mr={2} display="inline-block">
      <Icon name="eye" color="primary" />
      <Icon name="eye" color="success" />
      <Icon name="eye" color="warning" />
    </Boxer>
  </Story>
</Canvas>

## Unknown or `undefined` name case

<Description>
  If name is `undefined` or not found warning is sent and default circle icon is
  displayed.
</Description>

<Canvas>
  <Story name="Default icon">
    <Boxer mr={2} display="inline-block">
      <Icon name={undefined} color="red" />
      <Icon name="unknown" color="orange" />
    </Boxer>
  </Story>
</Canvas>

## Vertical Align

<Description>
  `Icon` is based on `Box`component which allow declarative css like
  `verticalAlign` in this example.
</Description>

<Canvas>
  <Story name="Vertical align">
    <Boxer display="flex">
      {['top', 'middle', 'bottom'].map(align => (
        <Box m={1}>
          <Icon name="eye" verticalAlign={align} /> {align}
        </Box>
      ))}
    </Boxer>
  </Story>
</Canvas>
