# Modal

## Basic Usage

<Canvas>
  <Story name="basic">
    {() => {
      const [opened, setOpened] = useState(false)
      return (
        <>
          <Button onClick={() => setOpened(!opened)}>Open Modal</Button>
          <Modal opened={opened} onClose={() => setOpened(false)}>
            <Box p={4}>Content should be present in center of the modal</Box>
          </Modal>
        </>
      )
    }}
  </Story>
</Canvas>
