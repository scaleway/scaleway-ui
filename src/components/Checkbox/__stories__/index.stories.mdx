# Checkbox

<Description>
  Checkbox control, it acts like a real checkbox.
</Description>

## Basic

<Canvas>
  <Story name="basic">
    {() => {
      const [checked, setChecked] = useState(false)
      const handleChange = useCallback(
        event => setChecked(event.target.checked),
        [],
      )
      return (
        <Checkbox checked={checked} onChange={handleChange}>
          Basic checkbox
        </Checkbox>
      )
    }}
  </Story>
</Canvas>
