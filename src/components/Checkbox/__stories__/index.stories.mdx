# Checkbox

<Description>Checkbox control, it acts like a real checkbox.</Description>

## Basic

<Canvas>
  <Story name="basic">
    {() => {
      const [checked, setChecked] = useState(false)
      const handleChange = useCallback(
        event => setChecked(event.target.checked),
        [],
      )
      return (
        <Boxer my={1}>
          <Checkbox checked={checked} onChange={handleChange}>
            Basic checkbox
          </Checkbox>
          <Checkbox checked disabled onChange={() => {}}>
            Basic disabled checkbox
          </Checkbox>
          <Checkbox checked={false} disabled onChange={() => {}}>
            Basic disabled checkbox not checked
          </Checkbox>
        </Boxer>
      )
    }}
  </Story>
</Canvas>
