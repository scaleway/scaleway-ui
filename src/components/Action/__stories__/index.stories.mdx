import {
  ArgsTable,
  Canvas,
  Description,
  Meta,
  Story,
} from '@storybook/addon-docs/blocks'
import Action from '..'
import { buttonVariants } from '../../Button'
import { icons } from '../../Icon'

<Meta title="Components/Action" component={Action} />

# Action

<Description>
  Action is an icon button that act as a native HTML `button`.
</Description>

<Canvas>
  <Story name="Basic">
    <Action name="lock" />
  </Story>
</Canvas>

## Variant

<Description>
  Using `variant` prop you can change the look and feel of the component.
</Description>

<Canvas>
  <Story name="Variant">
    {buttonVariants.map(variant => (
      <Action name="lock" variant={variant} mr={2} />
    ))}
  </Story>
</Canvas>

## Size

<Description>
  Set size using `size` prop. It will throw an error if it is greater than
  **32**.
</Description>

<Canvas>
  <Story name="Size">
    {[10, 18, 24, 32].map(size => (
      <Action key={size} name="lock" variant="secondary" size={size} mr={2} />
    ))}
  </Story>
</Canvas>

## Disabled

<Description>Disable the component with `disabled` prop.</Description>

<Canvas>
  <Story name="Disabled">
    <Action name="lock" mr={2} disabled />
  </Story>
</Canvas>

## Icon

<Description>You can change the icon by setting `name` prop.</Description>

<Canvas>
  <Story name="Icon">
    {icons.map(icon => (
      <Action key={icon} name={icon} mr={2} mb={2} />
    ))}
  </Story>
</Canvas>

## Rounded

<Description>Make the button rounded by adding `rounded` prop.</Description>

<Canvas>
  <Story name="Rounded">
    <Action name="lock" rounded />
  </Story>
</Canvas>

## Tooltip

<Description>Set tooltip using `tooltip` property.</Description>

<Canvas>
  <Story name="Tooltip">
    <Action name="lock" variant="primary" tooltip="I am locked" />
  </Story>
</Canvas>

## String

<Description>
  You can fill action button with text by adding it in `children`. You will also
  need to adjust width value.
</Description>

<Canvas>
  <Story name="String">
    <Action variant="secondary-bordered" width={32}>
      hello
    </Action>
  </Story>
</Canvas>
